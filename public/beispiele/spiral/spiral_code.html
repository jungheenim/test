let theta = 0.0, r = 0.0;

        function setup() {
          createCanvas(1000, 800);
          background(0);
        }
        
        function draw() {
          background(0, 1);
          let x = width / 2 + r * cos(theta);
          let y = height / 2 - r * sin(theta);
          rectMode(CENTER);
          stroke(255);
          //strokeWeight(10);
          noFill();
          //line(width / 2, height / 2, x, y);
          ellipse(x, y, r, r);
          theta += 0.05;
          r += 0.2;
        }
