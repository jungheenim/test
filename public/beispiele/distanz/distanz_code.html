function setup() {
          createCanvas(500, 500);
          background(0);
        }
        
        function draw() {
          background(0);
          let d = dist(width/2, height/2, mouseX, mouseY);
          stroke(255);
          strokeWeight(5);
          point(width/2, height/2);
          strokeWeight(1);
          fill(255);
          textSize(29);
          text(d, mouseX, mouseY);
          point(mouseX, mouseY);
          line(width/2, height/2, mouseX, mouseY);
        }
